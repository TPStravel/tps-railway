function getTransitionDay() {
  const startDate = localStorage.getItem('temarix_v3_end_date');
  if (!startDate) return 1;

  const diff = Math.floor((Date.now() - new Date(startDate)) / (1000 * 60 * 60 * 24)) + 1;
  return Math.min(diff, 7);
}
